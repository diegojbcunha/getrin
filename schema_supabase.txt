    -- Habilitar a extensão uuid-ossp para geração de UUIDs primários
    CREATE EXTENSION IF NOT EXISTS "uuid-ossp";

    -- 0. TABELA: EMPRESAS (companies)
    CREATE TABLE IF NOT EXISTS companies (
        id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
        name VARCHAR(255) NOT NULL,
        cnpj VARCHAR(20) UNIQUE,
        created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
    );

    -- Inserir empresa padrão para evitar erros de FK
    INSERT INTO companies (id, name) VALUES ('00000000-0000-0000-0000-000000000000', 'Empresa Padrão') ON CONFLICT DO NOTHING;

    -- 1. TABELA: TRABALHADORES (workers)
    CREATE TABLE IF NOT EXISTS public.workers ( 
       id uuid NOT NULL DEFAULT uuid_generate_v4 (), 
       company_id uuid NOT NULL, 
       name character varying(255) NOT NULL, 
       initials character varying(10) NOT NULL DEFAULT 'FT'::character varying, 
       matricula character varying(50) NOT NULL DEFAULT 'Aguardando'::character varying, 
       sector character varying(150) NOT NULL DEFAULT 'Geral'::character varying, 
       manager character varying(155) NULL, 
       admission date NULL, 
       email character varying(255) NULL, 
       phone character varying(50) NULL, 
       created_at timestamp with time zone NULL DEFAULT CURRENT_TIMESTAMP, 
       role character varying(150) NOT NULL DEFAULT 'Operador'::character varying, 
       compliance INTEGER DEFAULT 0 CHECK (compliance >= 0 AND compliance <= 100),
       status VARCHAR(50) DEFAULT 'gray',
       status_label VARCHAR(100) DEFAULT 'Pendente',
       CONSTRAINT workers_pkey PRIMARY KEY (id), 
       CONSTRAINT workers_company_id_email_key UNIQUE (company_id, email), 
       CONSTRAINT workers_company_id_matricula_key UNIQUE (company_id, matricula), 
       CONSTRAINT workers_company_id_fkey FOREIGN KEY (company_id) REFERENCES public.companies (id) ON DELETE CASCADE, 
       CONSTRAINT workers_id_fkey FOREIGN KEY (id) REFERENCES public.users_profile (id) 
    ) TABLESPACE pg_default;

    -- 2. TABELA: CATÁLOGO DE TREINAMENTOS (trainings)
    CREATE TABLE IF NOT EXISTS trainings (
        id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
        company_id UUID REFERENCES companies(id), -- NULL = biblioteca global
        name VARCHAR(255) NOT NULL,
        norm VARCHAR(50) NOT NULL,
        hours VARCHAR(50) NOT NULL,
        validity VARCHAR(50) NOT NULL,
        validity_months INTEGER DEFAULT 12, -- Novo campo para cálculos de trigger
        roles VARCHAR(255),
        mode VARCHAR(100) NOT NULL,
        status VARCHAR(50) DEFAULT 'green',
        status_label VARCHAR(100) DEFAULT 'Ativo',
        created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
    );

    -- 3. TABELA: ASSOCIAÇÃO DE TREINAMENTOS E TRABALHADORES (worker_trainings)
    CREATE TABLE IF NOT EXISTS worker_trainings (
        id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
        worker_id UUID NOT NULL REFERENCES workers(id) ON DELETE CASCADE,
        training_id UUID NOT NULL REFERENCES trainings(id) ON DELETE CASCADE,
        progress INTEGER DEFAULT 0 CHECK (progress >= 0 AND progress <= 100),
        done VARCHAR(50) DEFAULT '—',
        done_at DATE, -- Novo campo para disparar triggers de alerta
        expires VARCHAR(50) DEFAULT '—',
        expires_color VARCHAR(50) DEFAULT '',
        status VARCHAR(50) DEFAULT 'gray',
        status_label VARCHAR(100) DEFAULT 'Pendente',
        created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
        CONSTRAINT unique_worker_training UNIQUE (worker_id, training_id)
    );

    -- 4. TABELA: ALERTAS GERAIS (alerts)
    CREATE TABLE IF NOT EXISTS alerts (
        id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
        company_id UUID NOT NULL REFERENCES companies(id) DEFAULT '00000000-0000-0000-0000-000000000000',
        norm VARCHAR(50) NOT NULL,
        title VARCHAR(255) NOT NULL,
        count INTEGER NOT NULL DEFAULT 0,
        days INTEGER,
        days_until_expiry INTEGER, -- Novo campo usado pelo trigger
        level VARCHAR(50) NOT NULL,
        created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
    );

    -- 5. TABELA: PERFIL DE USUÁRIO (users_profile)
    CREATE TABLE IF NOT EXISTS users_profile (
        id UUID PRIMARY KEY REFERENCES auth.users(id) ON DELETE CASCADE,
        company_id UUID NOT NULL REFERENCES companies(id) DEFAULT '00000000-0000-0000-0000-000000000000',
        name VARCHAR(255),
        role VARCHAR(50) DEFAULT 'worker',
        created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
    );

    -- 6. TABELA: LOGS DE AUDITORIA (audit_logs)
    CREATE TABLE IF NOT EXISTS audit_logs (
        id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
        company_id UUID REFERENCES companies(id),
        user_id UUID REFERENCES auth.users(id),
        action VARCHAR(50) NOT NULL,
        table_name VARCHAR(100) NOT NULL,
        record_id UUID NOT NULL,
        created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
    );

    -- ============================================================
    -- FUNÇÕES E TRIGGERS (PL/pgSQL)
    -- ============================================================

    -- 1. Sincronização de Usuários (auth.users -> public.users_profile)
    CREATE OR REPLACE FUNCTION public.handle_new_user() 
    RETURNS TRIGGER AS $$ 
    BEGIN 
      INSERT INTO public.users_profile (id, company_id, name, role) 
      VALUES ( 
        NEW.id, 
        (COALESCE(NEW.raw_user_meta_data->>'company_id', '00000000-0000-0000-0000-000000000000'))::uuid, 
        COALESCE(NEW.raw_user_meta_data->>'name', 'Novo Usuário'), 
        COALESCE(NEW.raw_user_meta_data->>'role', 'worker') 
      ); 
      RETURN NEW; 
    END; 
    $$ LANGUAGE plpgsql SECURITY DEFINER; 

    CREATE OR REPLACE TRIGGER on_auth_user_created 
      AFTER INSERT ON auth.users 
      FOR EACH ROW EXECUTE FUNCTION public.handle_new_user(); 

    -- 2. Automação de Alertas de Vencimento
    CREATE OR REPLACE FUNCTION public.generate_training_alerts() 
    RETURNS TRIGGER AS $$ 
    DECLARE 
        v_company_id UUID; 
        v_norm VARCHAR; 
        v_validity_months INT; 
        v_expiry_date DATE; 
        v_days_until_expiry INT; 
    BEGIN 
        IF NEW.progress = 100 AND NEW.done_at IS NOT NULL THEN 
            SELECT t.company_id, t.norm, t.validity_months, w.company_id 
            INTO v_company_id, v_norm, v_validity_months, v_company_id 
            FROM public.trainings t 
            JOIN public.workers w ON w.id = NEW.worker_id 
            WHERE t.id = NEW.training_id; 

            v_expiry_date := NEW.done_at + (v_validity_months || ' months')::INTERVAL; 
            v_days_until_expiry := v_expiry_date - CURRENT_DATE; 

            IF v_days_until_expiry <= 60 THEN 
                INSERT INTO public.alerts (company_id, norm, title, days_until_expiry, level) 
                VALUES ( 
                    COALESCE(v_company_id, '00000000-0000-0000-0000-000000000000'), 
                    v_norm, 
                    'Treinamento próximo do vencimento ou vencido', 
                    v_days_until_expiry, 
                    CASE WHEN v_days_until_expiry <= 15 THEN 'urgent' ELSE 'monitor' END 
                ); 
            END IF; 
        END IF; 
        RETURN NEW; 
    END; 
    $$ LANGUAGE plpgsql SECURITY DEFINER; 

    CREATE OR REPLACE TRIGGER trg_generate_training_alerts 
      AFTER INSERT OR UPDATE ON public.worker_trainings 
      FOR EACH ROW EXECUTE FUNCTION public.generate_training_alerts(); 

    -- 3. Logs de Auditoria
    CREATE OR REPLACE FUNCTION public.process_audit_log() 
    RETURNS TRIGGER AS $$ 
    DECLARE 
        v_user_id UUID; 
        v_company_id UUID; 
        v_record_id UUID; 
        v_action VARCHAR; 
    BEGIN 
        BEGIN v_user_id := auth.uid(); EXCEPTION WHEN OTHERS THEN v_user_id := NULL; END; 

        IF (TG_OP = 'DELETE') THEN 
            v_action := 'DELETE'; v_record_id := OLD.id; 
            BEGIN v_company_id := OLD.company_id; EXCEPTION WHEN OTHERS THEN v_company_id := NULL; END; 
        ELSIF (TG_OP = 'UPDATE') THEN 
            v_action := 'UPDATE'; v_record_id := NEW.id; 
            BEGIN v_company_id := NEW.company_id; EXCEPTION WHEN OTHERS THEN v_company_id := NULL; END; 
        ELSIF (TG_OP = 'INSERT') THEN 
            v_action := 'INSERT'; v_record_id := NEW.id; 
            BEGIN v_company_id := NEW.company_id; EXCEPTION WHEN OTHERS THEN v_company_id := NULL; END; 
        END IF; 

        INSERT INTO public.audit_logs (company_id, user_id, action, table_name, record_id) 
        VALUES (v_company_id, v_user_id, v_action, TG_TABLE_NAME::character varying, v_record_id); 

        IF (TG_OP = 'DELETE') THEN RETURN OLD; ELSE RETURN NEW; END IF; 
    END; 
    $$ LANGUAGE plpgsql SECURITY DEFINER; 

    CREATE OR REPLACE TRIGGER trg_audit_workers AFTER INSERT OR UPDATE OR DELETE ON public.workers FOR EACH ROW EXECUTE FUNCTION public.process_audit_log(); 
    CREATE OR REPLACE TRIGGER trg_audit_trainings AFTER INSERT OR UPDATE OR DELETE ON public.trainings FOR EACH ROW EXECUTE FUNCTION public.process_audit_log(); 

    -- 4. Limpeza de Alertas
    CREATE OR REPLACE FUNCTION public.clean_expired_alerts() 
    RETURNS INT AS $$ 
    DECLARE deleted_count INT; 
    BEGIN 
        DELETE FROM public.alerts WHERE created_at < CURRENT_TIMESTAMP - INTERVAL '90 days'; 
        GET DIAGNOSTICS deleted_count = ROW_COUNT; 
        RETURN deleted_count; 
    END; 
    $$ LANGUAGE plpgsql SECURITY DEFINER;

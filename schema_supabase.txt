CREATE EXTENSION IF NOT EXISTS "uuid-ossp";

-- 0. TABELA: EMPRESAS (companies)
CREATE TABLE IF NOT EXISTS public.companies (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    name VARCHAR(255) NOT NULL,
    cnpj VARCHAR(20) UNIQUE NOT NULL,
    sector VARCHAR(255),
    contact_email VARCHAR(255),
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- 1. TABELA: PERFIL DE USUÁRIO (users_profile)
CREATE TABLE IF NOT EXISTS public.users_profile (
    id UUID PRIMARY KEY REFERENCES auth.users(id) ON DELETE CASCADE,
    company_id UUID REFERENCES public.companies(id),
    name VARCHAR(255),
    role VARCHAR(50) DEFAULT 'collaborator',
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- 2. TABELA: TRABALHADORES (workers)
CREATE TABLE IF NOT EXISTS public.workers (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    company_id UUID NOT NULL REFERENCES public.companies(id),
    name VARCHAR(255) NOT NULL,
    initials VARCHAR(10) NOT NULL DEFAULT 'FT',
    matricula VARCHAR(50) NOT NULL DEFAULT 'Aguardando',
    sector VARCHAR(150) NOT NULL DEFAULT 'Geral',
    manager VARCHAR(255),
    admission DATE,
    email VARCHAR(255),
    phone VARCHAR(50),
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    role VARCHAR(150) NOT NULL DEFAULT 'Operador',
    compliance INTEGER DEFAULT 0, -- Necessário para Dashboard
    status VARCHAR(50) DEFAULT 'gray', -- Necessário para Dashboard
    status_label VARCHAR(100) DEFAULT 'Pendente',
    CONSTRAINT workers_id_fkey FOREIGN KEY (id) REFERENCES public.users_profile(id)
);

-- 3. TABELA: CATÁLOGO DE TREINAMENTOS (trainings)
CREATE TABLE IF NOT EXISTS public.trainings (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    company_id UUID REFERENCES public.companies(id),
    name VARCHAR(255) NOT NULL,
    norm VARCHAR(50) NOT NULL,
    hours INTEGER NOT NULL,
    validity_months INTEGER NOT NULL DEFAULT 12,
    mode VARCHAR(100) NOT NULL,
    is_active BOOLEAN DEFAULT true,
    status VARCHAR(50) DEFAULT 'green',
    status_label VARCHAR(100) DEFAULT 'Ativo',
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- 4. TABELA: ASSOCIAÇÃO (worker_trainings)
CREATE TABLE IF NOT EXISTS public.worker_trainings (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    worker_id UUID NOT NULL REFERENCES public.workers(id) ON DELETE CASCADE,
    training_id UUID NOT NULL REFERENCES public.trainings(id) ON DELETE CASCADE,
    progress INTEGER DEFAULT 0 CHECK (progress >= 0 AND progress <= 100),
    done VARCHAR(50) DEFAULT '—', -- Adicionado para compatibilidade
    done_at DATE,
    expires VARCHAR(50) DEFAULT '—',
    status VARCHAR(50) DEFAULT 'gray',
    status_label VARCHAR(100) DEFAULT 'Pendente',
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT unique_worker_training UNIQUE (worker_id, training_id)
);

-- 5. TABELA: ALERTAS (alerts)
CREATE TABLE IF NOT EXISTS public.alerts (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    company_id UUID NOT NULL REFERENCES public.companies(id),
    norm VARCHAR(50) NOT NULL,
    title VARCHAR(255) NOT NULL,
    days_until_expiry INTEGER NOT NULL,
    level VARCHAR(50) NOT NULL CHECK (level IN ('urgent', 'monitor')),
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- 6. TABELA: LOGS DE AUDITORIA (audit_logs)
CREATE TABLE IF NOT EXISTS public.audit_logs (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    company_id UUID REFERENCES public.companies(id),
    user_id UUID REFERENCES auth.users(id),
    action VARCHAR(50) NOT NULL,
    table_name VARCHAR(100) NOT NULL,
    record_id UUID,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);
